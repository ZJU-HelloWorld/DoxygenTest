214BMI088ErrState BMI088::accInit(bool self_test) const
215{
216 DelayUs(kCommWaitTime);
217 accRead(BMI088_ACC_ACC_CHIP_ID); // 假读取
218
219 DelayUs(kCommWaitTime);
220 if (accRead(BMI088_ACC_ACC_CHIP_ID) != BMI088_ACC_ACC_CHIP_ID_VALUE) {
221 return kBMI088ErrStateAccNotFound;
222 }
223
224 if (self_test) {
225 if (accSelfTest() != kBMI088ErrStateNoErr) {
226 return kBMI088ErrStateAccSelfTestFailed;
227 }
228 }
229
230 // 配置一些寄存器
231 uint8_t acc_conf = (config_.acc_osr << _BMI088_ACC_ACC_CONF_ACC_OSR_SHFITS) |
232 (config_.acc_odr << _BMI088_ACC_ACC_CONF_ACC_ODR_SHFITS);
233 uint8_t acc_init_config[4][2] = {
234 {BMI088_ACC_ACC_RANGE, config_.acc_range},
235 {BMI088_ACC_ACC_CONF, acc_conf},
236 {BMI088_ACC_ACC_PWR_CTRL, BMI088_ACC_ACC_PWR_CTRL_ON},
237 {BMI088_ACC_ACC_PWR_CONF, BMI088_ACC_ACC_PWR_CONF_ACTIVE_MODE}};
238
239 DelayUs(kCommWaitTime);
240 accWrite(BMI088_ACC_ACC_SOFTRESET, BMI088_ACC_ACC_SOFTRESET_VALUE);
241 HAL_Delay(kAccResetTime);
242
243 DelayUs(kCommWaitTime);
244 accRead(BMI088_ACC_ACC_CHIP_ID); // 假读取
245
246 for (uint8_t i = 0; i < 4; i++) {
247 DelayUs(kCommWaitTime);
248 accWrite(acc_init_config[i][0], acc_init_config[i][1]);
249 DelayUs(kCheckWaitTime);
250 if (accRead(acc_init_config[i][0]) != acc_init_config[i][1]) {
251 return kBMI088ErrStateAccConfigErr;
252 }
253 }
254
255 return kBMI088ErrStateNoErr;
256}

305BMI088ErrState BMI088::accSelfTest(void) const
306{
307 uint8_t acc_init_config[4][2] = {
308 {BMI088_ACC_ACC_RANGE, BMI088_ACC_ACC_RANGE_ACC_RANGE_24G},
309 {BMI088_ACC_ACC_CONF, BMI088_ACC_ACC_CONF_NORMAL |
310 BMI088_ACC_ACC_CONF_ACC_ODR_1600_HZ},
311 {BMI088_ACC_ACC_PWR_CTRL, BMI088_ACC_ACC_PWR_CTRL_ON},
312 {BMI088_ACC_ACC_PWR_CONF, BMI088_ACC_ACC_PWR_CONF_ACTIVE_MODE}};
313
314 for (uint8_t i = 0; i < 4; i++) {
315 DelayUs(kCommWaitTime);
316 accWrite(acc_init_config[i][0], acc_init_config[i][1]);
317 DelayUs(kCheckWaitTime);
318 if (accRead(acc_init_config[i][0]) != acc_init_config[i][1]) {
319 return kBMI088ErrStateAccSelfTestFailed;
320 }
321 }
322
323 HAL_Delay(kAccSelfTestShortDelay);
324
325 float pos_self_test_data[3], neg_self_test_data[3];
326
327 accWrite(BMI088_ACC_ACC_SELF_TEST, BMI088_ACC_ACC_SELF_TEST_POSITIVE_SIGNAL);
328 HAL_Delay(kAccSelfTestLongDelay);
329 getAccData(pos_self_test_data);
330
331 accWrite(BMI088_ACC_ACC_SELF_TEST, BMI088_ACC_ACC_SELF_TEST_NEGATIVE_SIGNAL);
332 HAL_Delay(kAccSelfTestLongDelay);
333 getAccData(neg_self_test_data);
334
335 float min_diff_thres[3] = {kGravAcc, kGravAcc, 0.5 * kGravAcc};
336 for (uint8_t i = 0; i < 3; i++) {
337 if (pos_self_test_data[i] - neg_self_test_data[i] < min_diff_thres[i]) {
338 return kBMI088ErrStateAccSelfTestFailed;
339 }
340 }
341
342 return kBMI088ErrStateNoErr;
343}

350BMI088ErrState BMI088::gyroSelfTest(void) const
351{
352 // 配置一些寄存器
353 DelayUs(kCommWaitTime);
354 gyroWrite(BMI088_GYRO_GYRO_SELF_TEST, BMI088_GYRO_GYRO_SELF_TEST_TRIG_BIST);
355
356 uint8_t test_result;
357 while (1) {
358 DelayUs(kCommWaitTime);
359 test_result = gyroRead(BMI088_GYRO_GYRO_SELF_TEST);
360
361 if (test_result & BMI088_GYRO_GYRO_SELF_TEST_BIST_RDY) {
362 if (test_result & BMI088_GYRO_GYRO_SELF_TEST_BIST_FAIL) {
363 return kBMI088ErrStateGyroSelfTestFailed;
364 } else {
365 return kBMI088ErrStateNoErr;
366 }
367 }

412float BMI088::getTemp(void) const
413{
414 uint8_t temp_raw_data[kTempRawDataLen];
415 accMultiRead(BMI088_ACC_TEMP_MSB, kTempRawDataLen, temp_raw_data);
416
417 int16_t temp_raw =
418 static_cast<uint16_t>((temp_raw_data[0] << _BMI088_ACC_TEMP_MSB_SHIFTS) |
419 (temp_raw_data[1] >> _BMI088_ACC_TEMP_LSB_SHIFTS));
420
421 if (temp_raw > 1023) {
422 temp_raw -= 2048;
423 }
424
425 return temp_raw * kTempSens + kTempOffset;
426}

534static void DelayUs(uint16_t us)
535{
536 uint32_t reload = SysTick->LOAD;
537 uint32_t ticks = us * (SystemCoreClock / 1e6f);
538 uint32_t t_last = SysTick->VAL;
539
540 uint32_t t_now = 0;
541 uint32_t t_cnt = 0;
542
543 while (t_cnt < ticks) {
544 t_now = SysTick->VAL;
545 if (t_now != t_last) {
546 if (t_now < t_last) {
547 t_cnt += t_last - t_now;
548 } else {
549 t_cnt += reload - t_now + t_last;
550 }
551 t_last = t_now;
552 }
553 }
