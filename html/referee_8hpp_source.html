122template <typename T>
123bool Referee::encodeFrame(const T& data, const FrameCmdId frame_cmd_id, const uint16_t sub_content_id, const uint16_t receiver_id, uint8_t* p_frame, size_t& frame_length)
124{
125 bool process_result = false;
126 static_assert(sizeof(T) + kFramePartLengthExceptData <= kRefereeMaxBuffer, "Frame data size is too large.");
127 size_t data_index = kFramePartIndexData;
128 tx_info_.header.frame_header.sof = kRefereeFrameHeaderSof;
129 tx_info_.header.frame_header.data_length = sizeof(T);
130 tx_info_.header.frame_header.sequence_number++;
131 tx_info_.header.frame_cmd_id = frame_cmd_id;
132 if (frame_cmd_id == kFrameCmdIdInterAmongRobots) {
133 static_assert(sizeof(T) + kFramePartLengthExceptData + sizeof(InterFrameHeader) <= kRefereeMaxBuffer, "InterAmongRobots Frame data size is too large.");
134 tx_info_.header.frame_header.data_length += sizeof(InterFrameHeader);
135 data_index += sizeof(InterFrameHeader);
136 tx_info_.header.inter_frame_header.sub_content_id = sub_content_id;
137 tx_info_.header.inter_frame_header.sender_id = robot_performance_.data().robot_id;
138 if (sub_content_id == kUiSubCmdIdDeleteLayer || sub_content_id == kUiSubCmdIdDrawCharacter || sub_content_id == kUiSubCmdIdDrawGraphic1 || sub_content_id == kUiSubCmdIdDrawGraphic2 || sub_content_id == kUiSubCmdIdDrawGraphic5 || sub_content_id == kUiSubCmdIdDrawGraphic7) {
139 tx_info_.header.inter_frame_header.receiver_id = tx_info_.header.inter_frame_header.sender_id | 0x0100;
140 } else {
141 tx_info_.header.inter_frame_header.receiver_id = receiver_id;
142 }
143 }
144 memcpy(&tx_info_.frame_raw[data_index], &data, sizeof(T));
145 size_t _frame_length = data_index + sizeof(T) + sizeof(Crc16_t);
146 process_result = encodeSetCrc(tx_info_.frame_raw, _frame_length);
147 if (process_result) {
148 frame_length = _frame_length;
149 memcpy(p_frame, tx_info_.frame_raw, frame_length);
150 };
151
152 return process_result;
153};
