106void Buzzer::play()
107{
108 if (!is_playing_) {
109 return;
110 }
111
112 uint32_t tick = dsp::tick::GetTickMs();
113
114 /* 播放结束处理 */
115 if (kTuneListInfoPtr_->list[tune_idx_] == kTuneEnd) {
116 if (play_config_ == kPlayConfigLoopPlayback) {
117 tune_switch_ = true;
118 tune_idx_ = 0;
119 } else {
120 is_playing_ = last_is_playing_ = false;
121 tune_switch_ = false;
122 tune_idx_ = 0;
123 mute();
124 return;
125 }
126 }
127
128 /* 切换音符 */
129 if (tune_switch_) {
130 tune_switch_ = false;
131 uint32_t auto_reload = tune2AutoReload(kTuneListInfoPtr_->list[tune_idx_]);
132 float intensity_scale =
133 hello_world::Bound(kTuneListInfoPtr_->intensity_scale, 0, 1);
134 float duty = kMaxDuty * intensity_scale;
135 __HAL_TIM_SET_AUTORELOAD(kHtim_, auto_reload);
136 __HAL_TIM_SET_COMPARE(kHtim_, kChannel_, auto_reload * duty);
137 }
138
139 if (!last_is_playing_ && is_playing_) { // 刚开始播放
140 tune_start_tick_ = tick;
141 } else if (tick - tune_start_tick_ >= kTuneListInfoPtr_->tune_duration) {
142 tune_start_tick_ = tick;
143 tune_idx_++;
144
145 if (tune_idx_ >= kTuneListMaxLen) { // 音符溢出处理
146 is_playing_ = last_is_playing_ = false;
147 tune_switch_ = false;
148 tune_idx_ = 0;
149 mute();
150 return;
151 } else {
152 tune_switch_ = true;
153 }
154 }
155
156 last_is_playing_ = is_playing_;
157}

165uint32_t Buzzer::tune2AutoReload(Tune tune) const
166{
167 if (tune < kTuneC3 || tune >= kTuneRst) {
168 return 0;
169 } else {
170 return static_cast<uint32_t>(
171 // 440 * 2^((n-69)/12) Hz
172 kTimerFreq / 440.0f * powf(1.05946f, (tune)-69) - 1);
173 }
