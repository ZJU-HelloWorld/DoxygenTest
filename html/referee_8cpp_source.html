77bool Referee::processByte(const uint8_t byte)
78{
79 bool process_result = false;
80 switch (rx_info_.rx_status) {
81 case kRxStatusWaitingHeaderSof:
82 if (byte == kRefereeFrameHeaderSof) {
83 rx_info_.frame_raw[rx_info_.frame_data_index++];
84 rx_info_.rx_status = kRxStatusHeader;
85 } else {
86 restartDecodeFrame(1);
87 }
88 break;
89 case kRxStatusHeader:
90 rx_info_.frame_raw[rx_info_.frame_data_index++];
91 if (rx_info_.frame_data_index == kFramePartIndexCmdId) {
92 if (VerifyCrc8CheckSum(rx_info_.frame_raw, kFramePartLengthHeader)) {
93 rx_info_.frame_expect_length =
94 rx_info_.header.frame_header.data_length +
95 kFramePartLengthExceptData;
96 rx_info_.rx_status = kRxStatusCmdDataTail;
97 } else {
98 restartDecodeFrame(rx_info_.frame_data_index);
99 }
100 }
101 break;
102 case kRxStatusCmdDataTail:
103 rx_info_.frame_raw[rx_info_.frame_data_index++];
104 if (rx_info_.frame_data_index == rx_info_.frame_expect_length) {
105 if (VerifyCrc16CheckSum(rx_info_.frame_raw,
106 rx_info_.frame_expect_length)) {
107 process_result =
108 decodeRxPackage(&rx_info_.frame_raw[kFramePartIndexCmdId]);
109 }
110 restartDecodeFrame(kRefereeMaxBuffer);
111 }
112 break;
113 default:
114 restartDecodeFrame();
115 break;
116 }
117 return process_result;
118};
