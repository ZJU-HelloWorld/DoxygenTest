60State BasicPid::calc(const float *ref_ls, const float *fdb_ls,
61 const float *ffd_ls, float *out_ls)
62{
63 // 记录上次PID控制器的输出
64 datas_.last_ref = datas_.ref;
65 datas_.last_fdb = datas_.fdb;
66 datas_.last_out = datas_.out;
67 datas_.last_err = datas_.err;
68
69 // 更新本次PID控制器的输入
70 float ref = ref_ls != nullptr ? ref_ls[0] : 0;
71 ref = HandleAcross0(ref, datas_.last_ref, params_.across0);
72 datas_.ref = params_.setpointe_ramping.calc(ref, datas_.last_ref);
73 datas_.fdb = fdb_ls != nullptr ? fdb_ls[0] : 0;
74 datas_.ff = ffd_ls != nullptr ? ffd_ls[0] : 0;
75 datas_.err = datas_.ref - datas_.fdb;
76 datas_.err = params_.dead_band.calc(datas_.err);
77
78 // 获取时间间隔 单位：秒
79 // 如果间隔超过阈值将重置节点
80 uint32_t tick = dsp::tick::GetTickMs();
81 uint32_t interval = tick - datas_.last_tick;
82 if (interval <= 1) {
83 interval = 1;
84 }
85 if (params_.max_interval_ms && interval > params_.max_interval_ms) {
86 interval = 1;
87 if (params_.auto_reset) {
88 reset();
89 }
90 }
91 datas_.interval_s = interval / 1000.0f;
92
93 float out = datas_.ff;
94
95 // 计算比例项
96 datas_.prop = params_.kp * datas_.err;
97 out += datas_.prop;
98
99 // 计算积分项
100 if (params_.ki) {
101 float inte_err = params_.inte_trapezoidal.calc(datas_.err, datas_.last_err);
102 float k_icr = params_.inte_changing_rate.calc(inte_err);
103 float k_is = params_.inte_separation.calc(inte_err);
104 float k_iaw = params_.inte_anti_windup.calc(inte_err, datas_.inte);
105 datas_.inte +=
106 (k_icr * k_iaw) * (params_.ki * datas_.interval_s) * inte_err;
107 out += k_is * datas_.inte;
108 } else {
109 datas_.inte = 0;
110 }
111
112 // 计算微分项
113 if (params_.kd) {
114 float diff = params_.diff_previous.calc(datas_.fdb, datas_.last_fdb,
115 datas_.err, datas_.last_err);
116 diff *= (params_.kd / datas_.interval_s);
117
118 datas_.diff = params_.diff_filter.calc(diff, datas_.diff);
119 out += datas_.diff;
120 } else {
121 datas_.diff = 0;
122 }
123
124 // 传递输出
125 datas_.out = params_.out_limit.calc(out);
126 if (out_ls != nullptr) {
127 out_ls[0] = datas_.out;
128 }
129 // 记录时间
130 datas_.last_tick = tick;
131
132 if (ref_ls && fdb_ls && ffd_ls && out_ls) {
133 return kStateOk;
134 } else {
135 return kStateError;
136 }
137};

140void PidList::push(const BasicPidParams &params)
141{
142 PidNode *node = new PidNode(params);
143 if (empty()) { // 如果链表为空，头节点也是尾节点
144 head_ptr_ = tail_ptr_ = node;
145 } else {
146 tail_ptr_->next_ptr_ = node;
147 tail_ptr_ = node;
148 }
149 size_++;
150};
151void PidList::removeTail()
152{
153 if (empty()) {
154 return;
155 }
156 if (tail_ptr_ == head_ptr_) {
157 delete tail_ptr_;
158 head_ptr_ = tail_ptr_ = nullptr;
159 size_--;
160 return;
161 }
162
163 PidNode *rm_ptr_ = tail_ptr_;
164 PidNode *new_tail_ptr_ = head_ptr_;
165 while (new_tail_ptr_->next_ptr_ != tail_ptr_) {
166 new_tail_ptr_ = new_tail_ptr_->next_ptr_;
167 }
168 tail_ptr_ = new_tail_ptr_;
169 tail_ptr_->next_ptr_ = nullptr;
170 delete rm_ptr_;
171 size_--;
172 return;
173};
174void PidList::clear()
175{
176 while (!empty()) removeTail();
177};

250State MultiNodesPid::cascade_calc(const float *ref_ls, const float *fdb_ls,
251 const float *ffd_ls, float *out_ls)
252{
253 if (fdb_ls == nullptr) {
254 out_ = 0;
255 return kStateError;
256 }
257 State calc_state = kStateOk;
258 float ref = 0;
259 if (ref_ls) {
260 ref = ref_ls[0];
261 } else {
262 calc_state = kStateError;
263 }
264 float ff = 0;
265 if (ffd_ls) {
266 ff = ffd_ls[0];
267 } else {
268 calc_state = kStateError;
269 }
270
271 float out = 0;
272 float zero_ffd = 0;
273 const float *zero_ffd_ptr = &zero_ffd;
274
275 PidList::iterator ptr = pid_list_.begin();
276 uint32_t count = pid_list_.size();
277 for (size_t i = 0; i < count; i++) {
278 if (ptr == pid_list_.end()) {
279 break;
280 }
281 ptr->obj_.calc(&ref, fdb_ls + i, zero_ffd_ptr, &out);
282 ref = out;
283 ptr = ptr->next_ptr_;
284 }
285
286 out_ = out_limit_.calc(out + ff);
287 if (out_ls) {
288 out_ls[0] = out_;
289 } else {
290 calc_state = kStateError;
291 }
292
293 return calc_state;
294};

296State MultiNodesPid::parallel_calc(const float *ref_ls, const float *fdb_ls,
297 const float *ffd_ls, float *out_ls)
298{
299 if (ref_ls == nullptr || fdb_ls == nullptr) {
300 out_ = 0;
301 return kStateError;
302 }
303 State calc_state = kStateOk;
304 float ff = 0;
305 if (ffd_ls) {
306 ff = ffd_ls[0];
307 } else {
308 calc_state = kStateError;
309 }
310 float zero_ffd = 0;
311 const float *zero_ffd_ptr = &zero_ffd;
312 float out = 0;
313 float node_out = 0;
314 PidNode *ptr = pid_list_.begin();
315 uint32_t count = pid_list_.size();
316 for (size_t i = 0; i < count; i++) {
317 if (ptr == pid_list_.end()) {
318 break;
319 }
320 ptr->obj_.calc(ref_ls + i, fdb_ls + i, zero_ffd_ptr, &node_out);
321 out += node_out;
322 ptr = ptr->next_ptr_;
323 }
324 out_ = out_limit_.calc(out + ff);
325 if (out_ls) {
326 out_ls[0] = out_;
327 } else {
328 calc_state = kStateError;
329 }
330
331 return calc_state;
