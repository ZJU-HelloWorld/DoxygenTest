54A1::A1(uint8_t id, const OptionalParams& optinal_params)
55{
56 /* 变量检查 */
57#pragma region
58 HW_ASSERT(1 <= id && id <= 6, "Error id: %d", id);
59 HW_ASSERT(
60 optinal_params.dir == kDirFwd || optinal_params.dir == kDirRev,
61 "Error dir: %d", optinal_params.dir);
62 HW_ASSERT(
63 optinal_params.angle_range == kAngleRange0To2Pi ||
64 optinal_params.angle_range == kAngleRangeNegInfToPosInf ||
65 optinal_params.angle_range == kAngleRangeNegPiToPosPi,
66 "Error angle range: %d", optinal_params.angle_range);
67 HW_ASSERT(
68 optinal_params.input_type == kInputTypeRaw ||
69 optinal_params.input_type == kInputTypeTorq ||
70 optinal_params.input_type == kInputTypeCurr,
71 "Error input type: %d", optinal_params.input_type);
72 HW_ASSERT(
73 optinal_params.ex_redu_rat > 0,
74 "Error external reduction ration: %f", optinal_params.ex_redu_rat);
75 HW_ASSERT(
76 optinal_params.max_raw_input_lim > 0,
77 "Error max raw input limit: %f", optinal_params.max_raw_input_lim);
78 HW_ASSERT(
79 optinal_params.max_torq_input_lim > 0,
80 "Error max torque input limit: %f", optinal_params.max_torq_input_lim);
81 HW_ASSERT(
82 optinal_params.max_curr_input_lim > 0,
83 "Error max current input limit: %f", optinal_params.max_curr_input_lim);
84#pragma endregion
85
86 motor_info_ = kA1MotorBaseInfo;
87
88 /* 根据 ID 设定报文 ID */
89 motor_info_.tx_id = id <= 3 ? kTx1_3_ : kTx4_6_;
90 motor_info_.rx_id = kRx0_ + id;
91 motor_info_.id = id;
92
93 motor_info_.dir = optinal_params.dir;
94 motor_info_.angle_range = optinal_params.angle_range;
95 motor_info_.input_type = optinal_params.input_type;
96 motor_info_.angle_offset = optinal_params.angle_offset;
97 /* 转子端力矩限制 */
98 float rotor_torq_lim = motor_info_.torq_input_lim / motor_info_.redu_rat;
99 if (optinal_params.remove_build_in_reducer) {
100 motor_info_.redu_rat = optinal_params.ex_redu_rat;
101 } else {
102 motor_info_.redu_rat *= optinal_params.ex_redu_rat;
103 }
104
105 /* 计算输入限制 */
106 float max_raw_input = motor_info_.raw_input_lim;
107 if (optinal_params.max_raw_input_lim != std::numeric_limits<float>::max()) {
108 max_raw_input = hello_world::Min(max_raw_input, optinal_params.max_raw_input_lim);
109 }
110 float max_torq_input = rotor_torq_lim * motor_info_.redu_rat;
111 if (optinal_params.max_torq_input_lim != std::numeric_limits<float>::max()) {
112 max_torq_input = hello_world::Min(max_torq_input, optinal_params.max_torq_input_lim);
113 }
114 float max_curr_input = motor_info_.curr_input_lim;
115 if (optinal_params.max_curr_input_lim != std::numeric_limits<float>::max()) {
116 max_curr_input = hello_world::Min(max_curr_input, optinal_params.max_curr_input_lim);
117 }
118
119 max_raw_input = hello_world::Min(max_raw_input, torq2raw(max_torq_input));
120 max_raw_input = hello_world::Min(max_raw_input, curr2raw(max_curr_input));
121 motor_info_.raw_input_lim = max_raw_input;
122 motor_info_.torq_input_lim = raw2torq(max_raw_input);
123 motor_info_.curr_input_lim = raw2curr(max_raw_input);
124}

135State A1::encode(uint8_t tx_data[8], uint32_t tx_msg_std_id) const
136{
137 /* 变量检查 */
138#pragma region
139 HW_ASSERT(tx_data, "Error pointer");
140#pragma endregion
141
142 if (tx_msg_std_id != motor_info_.tx_id) {
143 return kStateIdDismatch;
144 } else {
145 uint8_t index = (motor_info_.id - 1) % 3;
146
147 int16_t input =
148 hello_world::Bound(static_cast<float>(motor_info_.dir) * raw_input_,
149 motor_info_.raw_input_lim, -motor_info_.raw_input_lim);
150 tx_data[2 * index] = input >> 8;
151 tx_data[2 * index + 1] = input;
152 tx_data[6] |= (0x01 << index);
153
154 return kStateOk;
155 }

166State A1::decode(const uint8_t rx_data[8], uint32_t rx_msg_std_id)
167{
168 /* 变量检查 */
169#pragma region
170 HW_ASSERT(rx_data, "Error pointer");
171#pragma endregion
172
173 if (rx_msg_std_id != motor_info_.rx_id) {
174 return kStateIdDismatch;
175 } else {
176 uint16_t raw_angle = static_cast<uint16_t>((rx_data[0] << 8) | rx_data[1]);
177
178 /* 判断是否旋转了一圈 */
179 float round = round_;
180 float delta_angle = raw_angle - last_raw_angle_;
181 if (fabsf(delta_angle) > motor_info_.cross_0_value * kCross0ValueThres) {
182 delta_angle < 0 ? round++ : round--;
183
184 /* 避免圈数溢出 */
185 if (motor_info_.angle_range != kAngleRangeNegInfToPosInf) {
186 round = fmodf(round, motor_info_.redu_rat);
187 }
188 }
189
190 float actual_ang =
191 static_cast<float>(motor_info_.dir) *
192 (raw_angle * motor_info_.angle_rat + round * 2 * PI) /
193 motor_info_.redu_rat;
194 float raw = static_cast<float>(motor_info_.dir) *
195 static_cast<int16_t>((rx_data[4] << 8) | rx_data[5]);
196
197 /* 统一对电机状态赋值 */
198 round_ = round;
199 actual_angle_ = normAngle(actual_ang);
200 angle_ = normAngle(actual_ang - motor_info_.angle_offset);
201 vel_ = static_cast<float>(motor_info_.dir) *
202 static_cast<int16_t>((rx_data[2] << 8) | rx_data[3]) *
203 motor_info_.vel_rat / motor_info_.redu_rat;
204 torq_ = raw2torq(raw);
205 curr_ = raw2curr(raw);
206 temp_ = rx_data[6];
207
208 last_raw_angle_ = raw_angle;
209
210 return kStateOk;
211 }
