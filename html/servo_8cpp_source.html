45Servo::Servo(TIM_HandleTypeDef* htim, uint32_t channel,
46 const ServoInfo& servo_info, float init_angle)
47 : kHtim_(htim), kChannel_(channel), kServoInfo_(servo_info)
48{
49 /* 变量检查 */
50#pragma region
51 HW_ASSERT(IS_TIM_INSTANCE(kHtim_->Instance), "Error TIM handle");
52 HW_ASSERT(IS_TIM_CHANNELS(kChannel_), "Error TIM channel: %d", kChannel_);
53 HW_ASSERT(kServoInfo_.operating_freq > 0, "Error Operating freqency: %.1f",
54 kServoInfo_.operating_freq);
55 HW_ASSERT(kServoInfo_.min_pulse_duration < kServoInfo_.max_pulse_duration &&
56 kServoInfo_.min_pulse_duration > 0u &&
57 kServoInfo_.max_pulse_duration <
58 static_cast<uint16_t>(1e6f / kServoInfo_.operating_freq),
59 "Error pulse duration");
60 HW_ASSERT(kServoInfo_.angle_range <= 360.0f,
61 "Error angle range: %d", kServoInfo_.angle_range);
62#pragma endregion
63
64 uint32_t auto_reload =
65 static_cast<uint32_t>(kTimerFreq / kServoInfo_.operating_freq) - 1;
66 __HAL_TIM_SET_AUTORELOAD(kHtim_, auto_reload);
67 __HAL_TIM_SET_COMPARE(kHtim_, kChannel_, angle2Cmp(init_angle));
68 HAL_TIM_PWM_Start(kHtim_, kChannel_);
69}

88uint32_t Servo::angle2Cmp(float angle) const
89{
90 angle = Bound(angle, 0.0f, kServoInfo_.angle_range);
91 uint32_t cmp = static_cast<uint32_t>(
92 angle *
93 (kServoInfo_.max_pulse_duration - kServoInfo_.min_pulse_duration) /
94 kServoInfo_.angle_range +
95 kServoInfo_.min_pulse_duration);
96
97 return cmp;
