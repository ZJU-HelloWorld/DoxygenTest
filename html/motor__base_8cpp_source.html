40State Motor::setInput(float input)
41{
42 float actual_input = 0;
43 switch (motor_info_.input_type) {
44 case kInputTypeRaw:
45 raw_input_ = hello_world::Bound(
46 input, motor_info_.raw_input_lim, -motor_info_.raw_input_lim);
47 torq_input_ = raw2torq(raw_input_);
48 curr_input_ = raw2curr(raw_input_);
49 actual_input = raw_input_;
50 break;
51 case kInputTypeTorq:
52 torq_input_ = hello_world::Bound(
53 input, motor_info_.torq_input_lim, -motor_info_.torq_input_lim);
54 raw_input_ = torq2raw(torq_input_);
55 curr_input_ = torq2curr(torq_input_);
56 actual_input = torq_input_;
57 break;
58 case kInputTypeCurr:
59 curr_input_ = hello_world::Bound(
60 input, motor_info_.curr_input_lim, -motor_info_.curr_input_lim);
61 raw_input_ = curr2raw(curr_input_);
62 torq_input_ = curr2torq(torq_input_);
63 actual_input = curr_input_;
64 break;
65 default:
66 return kStateInputTypeError;
67 }
68
69 if (fabsf(actual_input) < fabsf(input)) {
70 return kStateInputValueOverflow;
71 } else {
72 return kStateOk;
73 }
74}

104float Motor::raw2torq(float raw) const
105{
106 if (motor_info_.raw_mapping_type == kRawMappingTypeTorq) {
107 return raw * motor_info_.torq_rat * motor_info_.redu_rat;
108 } else if (motor_info_.raw_mapping_type == kRawMappingTypeCurr) {
109 return raw * motor_info_.curr_rat * motor_info_.torq_const *
110 motor_info_.redu_rat;
111 } else {
112 return 0;
