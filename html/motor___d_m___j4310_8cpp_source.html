65static inline float _Uint2Float(
66 uint16_t x_int, float x_max, float x_min, uint8_t bits)
67{
68 float span = x_max - x_min;
69 float offset = x_min;
70 return ((float)x_int) * span / ((float)((1 << bits) - 1)) + offset;
71}

82static inline uint16_t _Float2Uint(
83 float x_float, float x_max, float x_min, uint8_t bits)
84{
85 float span = x_max - x_min;
86 float offset = x_min;
87 return (uint16_t)((x_float - offset) * ((float)((1 << bits) - 1)) / span);
88}

97DM_J4310::DM_J4310(uint8_t id, const OptionalParams& optinal_params)
98{
99 /* 变量检查 */
100#pragma region
101 HW_ASSERT(1 <= id && id <= 10, "Error id: %d", id);
102 HW_ASSERT(
103 optinal_params.dir == kDirFwd || optinal_params.dir == kDirRev,
104 "Error dir: %d", optinal_params.dir);
105 HW_ASSERT(
106 optinal_params.angle_range == kAngleRange0To2Pi ||
107 optinal_params.angle_range == kAngleRangeNegInfToPosInf ||
108 optinal_params.angle_range == kAngleRangeNegPiToPosPi,
109 "Error angle range: %d", optinal_params.angle_range);
110 HW_ASSERT(
111 optinal_params.input_type == kInputTypeRaw ||
112 optinal_params.input_type == kInputTypeTorq ||
113 optinal_params.input_type == kInputTypeCurr,
114 "Error input type: %d", optinal_params.input_type);
115 HW_ASSERT(
116 optinal_params.ex_redu_rat > 0,
117 "Error external reduction ration: %f", optinal_params.ex_redu_rat);
118 HW_ASSERT(
119 optinal_params.max_torq_input_lim > 0,
120 "Error max torque input limit: %f", optinal_params.max_torq_input_lim);
121 HW_ASSERT(
122 optinal_params.max_curr_input_lim > 0,
123 "Error max current input limit: %f", optinal_params.max_curr_input_lim);
124#pragma endregion
125
126 motor_info_ = kDM_J4310MotorBaseInfo;
127
128 /* 根据 ID 设定报文 ID */
129 motor_info_.tx_id = kTx0_ + id;
130 motor_info_.rx_id = kRx0_ + id;
131 motor_info_.id = id;
132
133 motor_info_.dir = optinal_params.dir;
134 motor_info_.angle_range = optinal_params.angle_range;
135 motor_info_.input_type = optinal_params.input_type;
136 motor_info_.angle_offset = optinal_params.angle_offset;
137 /* 转子端力矩限制 */
138 float rotor_torq_lim = motor_info_.torq_input_lim / motor_info_.redu_rat;
139 if (optinal_params.remove_build_in_reducer) {
140 motor_info_.redu_rat = optinal_params.ex_redu_rat;
141 } else {
142 motor_info_.redu_rat *= optinal_params.ex_redu_rat;
143 }
144
145 /* 计算输入限制 */
146 float max_torq_input = rotor_torq_lim * motor_info_.redu_rat;
147 if (optinal_params.max_torq_input_lim != std::numeric_limits<float>::max()) {
148 max_torq_input = hello_world::Min(max_torq_input, optinal_params.max_torq_input_lim);
149 }
150 float max_curr_input = motor_info_.curr_input_lim;
151 if (optinal_params.max_curr_input_lim != std::numeric_limits<float>::max()) {
152 max_curr_input = hello_world::Min(max_curr_input, optinal_params.max_curr_input_lim);
153 }
154
155 max_torq_input = hello_world::Min(max_torq_input, kMaxTorq);
156 max_torq_input = hello_world::Min(max_torq_input, curr2torq(max_curr_input));
157 motor_info_.raw_input_lim = kInvalidValue;
158 motor_info_.torq_input_lim = max_torq_input;
159 motor_info_.curr_input_lim = torq2curr(max_torq_input);
160
161 is_start_ = false;
162}

164State DM_J4310::setInput(float input)
165{
166 float torq_input = 0;
167 switch (motor_info_.input_type) {
168 case kInputTypeRaw:
169 torq_input = raw2torq(input);
170 torq_input_ = hello_world::Bound(
171 input, motor_info_.torq_input_lim, -motor_info_.torq_input_lim);
172 curr_input_ = raw2torq(raw_input_);
173 raw_input_ = torq2raw(torq_input_);
174 if (fabsf(torq_input_) < fabsf(torq_input)) {
175 return kStateInputValueOverflow;
176 } else {
177 return kStateOk;
178 }
179 case kInputTypeTorq:
180 torq_input_ = hello_world::Bound(
181 input, motor_info_.torq_input_lim, -motor_info_.torq_input_lim);
182 raw_input_ = torq2raw(torq_input_);
183 curr_input_ = torq2curr(torq_input_);
184 if (fabsf(torq_input_) < fabsf(input)) {
185 return kStateInputValueOverflow;
186 } else {
187 return kStateOk;
188 }
189 case kInputTypeCurr:
190 curr_input_ = hello_world::Bound(
191 input, motor_info_.curr_input_lim, -motor_info_.curr_input_lim);
192 raw_input_ = curr2raw(curr_input_);
193 torq_input_ = curr2torq(torq_input_);
194 if (fabsf(curr_input_) < fabsf(input)) {
195 return kStateInputValueOverflow;
196 } else {
197 return kStateOk;
198 }
199 default:
200 return kStateInputTypeError;
201 }

213State DM_J4310::encode(uint8_t tx_data[8], uint32_t tx_msg_std_id) const
214{
215 /* 变量检查 */
216#pragma region
217 HW_ASSERT(tx_data, "Error pointer");
218#pragma endregion
219
220 if (tx_msg_std_id != motor_info_.tx_id) {
221 return kStateIdDismatch;
222 } else {
223 if (is_start_) {
224 uint16_t input =
225 hello_world::Bound(static_cast<uint16_t>(raw_input_),
226 static_cast<uint16_t>((0x01 << kTorqBits) - 1),
227 static_cast<uint16_t>(0));
228 memset(tx_data, 0x00, sizeof(uint8_t) * 7);
229 tx_data[6] |= (input >> 8) & 0x0F;
230 tx_data[7] = input & 0xFF;
231 } else {
232 memset(tx_data, 0xFF, sizeof(uint8_t) * 7);
233 tx_data[7] = 0xFC;
234 }
235
236 return kStateOk;
237 }

248State DM_J4310::decode(const uint8_t rx_data[8], uint32_t rx_msg_std_id)
249{
250 /* 变量检查 */
251#pragma region
252 HW_ASSERT(rx_data, "Error pointer");
253#pragma endregion
254
255 if (rx_msg_std_id != motor_info_.rx_id) {
256 return kStateIdDismatch;
257 } else {
258 is_start_ = true;
259 uint16_t raw_angle = static_cast<uint16_t>((rx_data[1] << 8) | rx_data[2]);
260 float angle = _Uint2Float(raw_angle, kMaxAngle, -kMaxAngle, kAngleBits);
261 float last_angle = _Uint2Float(
262 last_raw_angle_, kMaxAngle, -kMaxAngle, kAngleBits);
263
264 /* 判断是否旋转了一圈 */
265 float round = round_;
266 float delta_angle = angle - last_angle;
267 if (fabsf(delta_angle) > 2 * PI * kCross0ValueThres) {
268 delta_angle < 0 ? round++ : round--;
269
270 /* 避免圈数溢出 */
271 if (motor_info_.angle_range != kAngleRangeNegInfToPosInf) {
272 round = fmodf(round, motor_info_.redu_rat);
273 }
274 }
275
276 float actual_ang =
277 static_cast<float>(motor_info_.dir) * (angle + round * 2 * PI) /
278 motor_info_.redu_rat;
279 float raw = static_cast<uint16_t>((rx_data[4] << 8) | rx_data[5]);
280
281 /* 统一对电机状态赋值 */
282 round_ = round;
283 actual_angle_ = normAngle(actual_ang);
284 angle_ = normAngle(actual_ang - motor_info_.angle_offset);
285 vel_ = static_cast<float>(motor_info_.dir) *
286 _Uint2Float(static_cast<int16_t>((rx_data[2] << 8) | rx_data[3]),
287 kMaxVel, -kMaxVel, kVelBits) /
288 motor_info_.redu_rat;
289 torq_ = static_cast<float>(motor_info_.dir) * raw2torq(raw);
290 curr_ = static_cast<float>(motor_info_.dir) * raw2curr(raw);
291 temp_ = rx_data[6];
292
293 last_raw_angle_ = raw_angle;
294
295 return kStateOk;
296 }
